

CREATE TABLE customer(name varchar,user_id numeric);

CREATE TABLE food_order(order_id varchar, user_id numeric, food_item varchar, order_date timestamp);

SELECT c.user_id, f.food_item
FROM customer as c
JOIN food_order as f
ON c.user_id=f.user_id
WHERE c.user_id=1;

SELECT c.user_id, f.order_date
FROM customer as c
JOIN food_order as f
ON c.user_id=f.user_id
WHERE f.order_date BETWEEN '2017-07-24 13:10:11' AND '2017-07-25 13:10:11';
